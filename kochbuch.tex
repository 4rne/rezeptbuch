    bakingtime={\unit[30-45]{min}},
    bakingtemperature={\unit[200-250]{\textcelcius}},
    source = Antje
]
{Kartoffel-Lachs-Auflauf}
    
    %\graph
    %{% Bilder
    %    small=pic/glass,    % kleines Bild
    %    big=pic/ingredients % großes (längeres) Bild
    %}
    
    \ingredients
    {% Zutaten
	\unit[1,5]{kg} & Kartoffeln \\
	\unit[200]{g} & Lachs \\
	\unit[200]{ml} & Sahne \\
	\unit[200]{ml} & Milch \\
	2 & Eier \\
	1 & Bund Dill \\
	& Pfeffer, Salz \\
	1 & Zitrone (optional)\\
    }
    
    \preparation
    { % Zubereitung
	\step Backofen vorheizen
	\step Pellkartoffeln kochen, pellen, in Scheiben schneiden. Lachs in Streifen schneiden.
	\step Kartoffeln und Lachs dachziegelartig in eine gefettete Auflaufform legen.
	\step Sauce aus Sahne, Milch, Eiern, Dill, Salz und Pfeffer darübergeben.
	\step Eine Dreiviertelstunde bei 200°-250°C backen.
    }
    
	\suggestion[Variante: Nudel-Lachs-Auflauf]
    {%
	Statt Kartoffeln 500g gekochte Nudeln verwenden und nur eine halbe Stunde backen, dafür die Sauce mit Milch verlängern.
    }
    
    \hint
    {% Tipp
      Mit Salat servieren
    }

\end{recipe}

\begin{recipe}
[% 
    preparationtime = {\unit[1,5]{h}},
    bakingtime={\unit[35-45]{min}},
    bakingtemperature={\unit[210-230]{\textcelcius}},
    portion = {\portion{4}},

]
{Quiche lorraine}
    
    %\graph
    %{% pictures
    %    small=pic/glass,     % small picture
    %    big=pic/ingredients  % big picture
    %}
    
    \ingredients{%
	\unit[200]{g} & Mehl \\
	\unit[150]{g} & Butter \\
	\unit[\nicefrac{1}{2}]{TL} & Salz \\
	\unit[3-4]{EL} & Wasser \\
	\multicolumn{2}{l}{\textbf{Belag:}} \\
	\unit[150]{g} & gekochter Schinken \\
	\unit[150]{g} & Emmentaler oder Gouda \\
	4 & Eier \\
	\unit[125]{ml} & saure Sahne \\
	\unit[\nicefrac{1}{2}]{TL} & Paprikapulver \\
	& etwas Petersilie \\
    }
    
    \preparation{%
	\step Backofen vorheizen.
	\step Das Mehl in eine Schüssel geben, das Wasser mit etwas Mehl verrühren. 
	Salz zufügen und die Buttern in Flöckchen darauf verteilen. 
	Alles zu einem glatten Teig verkneten und 1 Stunde ruhen lassen. 
	\step Den Teig ausrollen und in eine gefettete Springform (\unit[24-26]{cm} \diameter) geben; einen \unit[3]{cm} hohen Rand andrücken. 
	Die Sahne mit den Eiern verquirlen und den feingewürfelten Schinken und Käse mit Paprika und gehackter Petersilie unterrühren. 
	Die Masse auf den Teigboden verteilen.
	\step 35-45 Minuten backen.
    }

\begin{recipe}
[% 
    preparationtime = {\unit[40]{min}},
    bakingtime={\unit[20]{min}},
    bakingtemperature={\unit[220]{\textcelcius}},
    portion = {\portion{4}},
    %calory={\unit[3]{kJ}},
    source = {\url{http://www.chefkoch.de/rezepte/837601188560864/Kartoffelgratin.html}}
]
{Kartoffelgratin}
    \ingredients{
	\unit[500]{g} & Kartoffeln \\
	\unit[125]{ml} & Milch \\
	\unit[125]{ml} & Sahne \\
	1 & Knoblauchzehe \\
	& Salz \\
	& Pfeffer \\
	& Rosmarin \\
	& Butter \\
	\unit[3]{EL} & Käse \\
    }
    
    \preparation{
    \step Die Kartoffeln schälen und in dünne Scheiben schneiden.

    \step Milch und Sahne vermischen, mit gepresstem Knoblauch, etwas Salz und Pfeffer sowie Rosmarin würzen und mit den Kartoffelscheiben sanft aufkochen lassen, dann unter häufigem Rühren ca. 10 Minuten köcheln lassen.
    
    \step Eine Auflaufform mit Butter bestreichen, Kartoffelmasse einfüllen, mit Käse bestreuen und im vorgeheiztem Ofen bei \unit[220]{\textcelcius} ca. 15 - 20 Minuten überbacken. 
    }
    
\end{recipe}

\begin{recipe}
[% 
    preparationtime = {\unit[30]{min}},
    bakingtime={\unit[15]{min}},
    bakingtemperature={\unit[200]{\textcelcius}},
    portion = {\pieces{20}},
    %calory={\unit[3]{kJ}},
  source = {\url{http://www.chefkoch.de/rezepte/67481024847665/Schnelle-Kaesestangen.html}}
]
{Käsestangen}
    
    %\graph
    %{% pictures
    %    small=pic/glass,     % small picture
    %    big=pic/ingredients  % big picture
    %}
    
    \ingredients{%
	\unit[6]{Platten} & Blätterteig \\
	2 & Eier \\
	\unit[400]{g} & Käse (z.B. Emmentaler) \\
	& Salz \\
	& Pfeffer \\
	& Basilikum \\
	& Oregano \\
    }
    
    \preparation{%
	\step Den Blätterteig auftauen lassen.
	
	\step Währenddessen den Käse reiben und würzen. Dann mit Ei vermischen.
	
	\step Eine Platte Blätterteig auf etwa die doppelte Größe ausrollen und ein Viertel der Käse-Masse darauf verteilen. 
	Eine weitere Platte Blätterteig darauflegen und erneut mit Füllung belegen. 
	Die letzte Blätterteigplatte darauf legen und alles auf die Größe der untersten Platte ausrollen. 
	
	\step Dünne Streifen von dem Stapel abschneiden, diese einmal verdehen und mit etwas Abstand auf ein Backblech legen.
	
	\step Bei \unit[200]{\textcelcius} - \unit[220]{\textcelcius} etwa \unit[15]{Minuten} backen.
    }
    
\end{recipe}

\begin{recipe}
[% 
    preparationtime = {\unit[45]{min}},
    bakingtime={\unit[25]{min}},
    bakingtemperature={\unit[180]{\textcelcius}},
    portion = {\portion{8}},
    %calory={\unit[3]{kJ}},
  source = {\url{Chefkoch}}
]
{Italienischer Nudelauflauf}
    
    %\graph
    %{% pictures
    %    small=pic/glass,     % small picture
    %    big=pic/ingredients  % big picture
    %}
    
    \ingredients{%
	\unit[500]{g} & Rinderhack \\
	\unit[400]{g} & geschälte gehackte Tomaten \\
	1 & Zwiebel \\
	\unit[2]{TL} & gehacktes Basilikum \\
	\unit[2]{Kugeln} & Mozarella \\
	\unit[500]{g} & Nudeln (Farfalle od. Spirelli) \\
	\unit[200]{g} & geriebenen Emmentaler \\
	2 & Zucchini \\
	1 & Knoblauchzehe \\
	\unit[\nicefrac{1}{2}]{Tube} & Tomatenmark \\
	\unit[2]{TL} & Gemüsebrühpulver  mit etwas Wasser\\
	\unit[2]{TL} & Sambal Oelek \\
	& Salz \\
    }
    
    \preparation{%
	\step Das Hackfleisch mit der gewürfelten Zwiebel scharf anbraten. 
	Mit in etwas Wasser aufgelösten Brühwürfeln ablöschen. 
	Tomaten, Basilikum und Knoblauch hinzufügen. Ca. 25 Minuten köcheln lassen. 
	\step Dann die Zucchini in geviertelte Scheiben geschnitten hinzufügen. 
	Weitere 10 Minuten köcheln lassen (die Zucchinistücke sollen nicht zu weich werden!) 
	
	\step In der Zeit die Nudeln al dente kochen und anschließend in eine Auflaufform geben. 
	Den Mozzarella kleinschneiden und unter die Nudeln heben. 
	
	\step Das Hackfleisch sehr würzig mit Salz abschmecken. 
	Tomatenmark hinzufügen und das Ganze mit Sambal Oelek würzen.
	
	\step Anschließend die Fleischsoße zu den Nudeln in die Auflaufform geben und alles vermengen. 
	Emmentaler darüber streuen.
	
	\step Bei \unit[180]{\textcelcius} im vorgeheizten Backofen \unit[25]{Minuten} backen.
    }
    
\end{recipe}

\begin{recipe}
[% 
    preparationtime = {\unit[80]{min}},
    bakingtime={\unit[12]{min}},
    bakingtemperature={\unit[200]{°C}},
    portion = {\pieces{12 - 15}},
    source = {Arnes Rezept nach \url{www.hamburg.de/franzbroetchen}}
]
{Hamburger Franzbrötchen}
    
    %\graph
    %{% pictures
    %    small=pic/glass,     % small picture
    %    big=pic/ingredients  % big picture
    %}
    
    \ingredients{%
	\unit[500]{g} & Mehl \\
	\unit[70]{g} & Zucker \\
	\unit[70]{g} & weiche Butter \\
	\unit[250]{ml} & lauwarme Milch \\
	& Würfel Hefe \\
	& Prise Salz \\
	\multicolumn{2}{l}{\textbf{Füllung}} \\
	\unit[180]{g} & weiche Butter \\
	\unit[4-5]{TL} & Zimt \\
	\unit[300]{g} & Zucker \\
    }
    
    \preparation{%
	\step Aus den Zutaten für den Teig einen Hefeteig herstellen, gut durchkneten und auf einer bemehlten Arbeitsfläche zu einem Rechteck ausrollen (ca. 30 x 25 cm). 
	
	\step \unit[80]{g} Butter gleichmäßig auf einer Teighälfte verstreichen, die andere Teighälfte darüber klappen und dabei achtgeben, dass sich keine Blasen im dem Teig bilden. 
	Dann die Teigränder zusammendrücken, Ränder unter den Teig schieben. Den Teig auf ca. 50 x 30 cm ausrollen. 
	Ein Drittel des Teiges von links zur Mitte hin einschlagen, das restliche Drittel von rechts darüber schlagen, sodass drei Schichten Teig übereinander liegen. 
	Auch hier ist darauf zu achten, dass keine Luftblasen im Teig entstehen.
	
	\step Den Teig mindestens 20 Minuten kühl stellen. Währenddessen für die Füllung Zimt und Zucker mit der restlichen in einem Topf geschmolzenen Butter verrühren. 
	
	\step Den kalten Teig auf ca. 80 x 40 cm ausrollen. Die Füllung gleichmäßig auf den unteren drei Vierteln des Teiges verteilen. 
	Von unten her vorsichtig den Teig von der Arbeitsfläche lösen und eng aufrollen. 
	\step Die Rolle in ca. 4 cm breite Trapeze schneiden und die Franzbrötchen in einem Abstand von mindestens \unit[10]{cm} mit der breiten Grundfläche des Trapezes auf ein Backblech legen.
	Dort mit dem Stiel eines Kochlöffels die Mitte fast ganz runterdrücken, sodass der Teig sich über dem Stiel zusammenwölbt und die Füllung und der Teig aus der Mitter herausquillt.
	
	\step Den Backofen auf \unit[200]{°C} vorheizen und die Franzbrötchen abgedeckt mindestens 20 Minuten gehen lassen. (Besser bei 50°C im Backofen)
	Nach dem Gehen die Franzbrötchen vorsichtig platt drücken.
	\step Im vorgeheizten Backofen bei Umluft ca. 12 Minuten backen, bis die typische Bräune erreicht ist.
    }
  
%    \hint{%
%        Warm genießen.
%    }
    
\end{recipe}
